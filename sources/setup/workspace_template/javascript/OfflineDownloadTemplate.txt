    <script type="text/javascript">
        // EaglerCraft X Options (Offline)
        window.eaglercraftXOpts = {
            container: "eaglercraft",
            assetsURI: "data:application/epk;base64,{{EAGLERCRAFT_ASSETS}}",
            localesURI: null,
            worldsDB: "worlds_offline",
            resourcePacksDB: "resourcePacks_offline",
            servers: [],
            relays: [],
            enableMinceraft: true,
            crashOnUncaughtExceptions: true,
            allowBootMenu: false,
            allowUpdateSvc: false,
            enableServerListing: false,
            offlineMode: true
        };
        
        // Loading Progress
        var _progress = 0;
        function setLoadingProgress(p, t, d) {
            _progress = Math.max(0, Math.min(100, p));
            var f = document.getElementById("progress-fill");
            var te = document.getElementById("progress-text");
            var de = document.getElementById("progress-detail");
            if (f) f.style.width = _progress + "%";
            if (te) te.textContent = t || "";
            if (de) de.textContent = d || "";
        }
        function hideLoadingScreen() {
            var s = document.getElementById("loading-screen");
            if (s) { s.classList.add("fade-out"); setTimeout(function(){ s.style.display = "none"; }, 500); }
        }
        function showError(m, d) {
            var o = document.getElementById("error-overlay");
            var me = document.getElementById("error-message");
            var de = document.getElementById("error-details");
            if (me) me.textContent = m;
            if (de) de.textContent = d || "";
            if (o) o.classList.add("visible");
        }
        
        // WebGL2 Check
        try {
            var _c = document.createElement("canvas");
            var _gl = _c.getContext("webgl2") || _c.getContext("experimental-webgl2");
            if (!_gl) {
                showError("WebGL2 is not supported!", "Please use a modern browser with WebGL2 support.");
            }
        } catch(e) {
            showError("WebGL2 initialization failed", e.message);
        }
        
        // Prevent defaults
        document.addEventListener("contextmenu", function(e) { e.preventDefault(); });
        document.addEventListener("dragover", function(e) { e.preventDefault(); });
        document.addEventListener("drop", function(e) { e.preventDefault(); });
        
        // Set canvas size
        var canvas = document.getElementById("eaglercraft");
        if (canvas) { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
        window.addEventListener("resize", function() {
            if (canvas) { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
        });
        
        setLoadingProgress(10, "Loading offline client...", "Bundled JS");
    </script>
